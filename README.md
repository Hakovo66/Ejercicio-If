# Ejercicio-If

public class  EjercicioIf {

    public static void main(String[] args) {

        int numeroIf = -1000;

        if (numeroIf == 0){
            System.out.println("el numeroIf es 0");

        } else if (numeroIf <= 0) {
            System.out.println ("el numeroIf es negativo");

        } else if (numeroIf >= 0) {
            System.out.println ("el numeroIf es positivo");

        }
    }


    
}
